\Psionic{Inflict Pain}{inflict pain}
{Telepathy [Mind-Affecting, Evil]}
{
	\textbf{Level:}
	Telepath 5\\
	\textbf{Manifesting Time:}
	1 round\\
	\textbf{Range:}
	Unlimited\\
	\textbf{Target:}
	One \emph{contacted} mind\\
	\textbf{Saving Throw:}
	Will partial\\
	\textbf{Cost:}
	\psionic{contact}\\
	\textbf{Maintenance Cost:}
	5 pp/round\\
	\textbf{Prerequisites:}
	\psionic{mindlink}\\
	% \textbf{Critical Success:}
	% The target becomes unconscious for 1d10 rounds..\\
	% \textbf{Critical Failure:}
	% This contact is broken.\\
}
{
	You telepathically stab the mind of your foe, causing horrible agony. The subject suffers wracking pain that imposes a $-4$ penalty on attack rolls, skill checks, and ability checks. If the target makes its save, it takes only a $-2$ penalty.
}
