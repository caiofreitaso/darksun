\Psionic{Heighten Senses}{heighten senses}
{Metacreativity}
{
	\textbf{Level:}
	Shaper 1\\
	\textbf{Manifesting Time:}
	1 mental action\\
	\textbf{Range:}
	12 meters\\
	\textbf{Area:}
	12-m-radius emanation centered on you\\
	\textbf{Saving Throw:}
	None\\
	\textbf{Cost:}
	5 power points\\
	\textbf{Maintenance Cost:}
	1 pp/round\\
	\textbf{Prerequisites:}
	\psionic{animal affinity}, \psionic{synesthete}\\
	% \textbf{Critical Success:}
	% One of the senses, chosen at random, stays with you for a full day.\\
	% \textbf{Critical Failure:}
	% You lose all your senses for 1d12 hours.\\
}
{
	You sharpen all five of your normal senses. You gain the following abilities:
	\begin{itemize*}
	\item You gain darkvision, low-light vision, and scent.
	\item You gain +10 bonus on \skill{Listen} and \skill{Spot} checks.
	\item You can taste poison and other impurities in any quantity.
	\item You can identify anything by touch. For example, you can tell two ceramic pieces from each other, after having previously handled just one of them. 
	\end{itemize*}
}
