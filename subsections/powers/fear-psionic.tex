\Psionic{Fear, Psionic}{psionic fear}
{Telepathy [Fear, Mind-Affecting]}
{
    \textbf{Level:}
    Telepath 4\\
    \textbf{Manifesting Time:}
    1 mental action\\
    \textbf{Range:}
    Unlimited\\
    \textbf{Target:}
    One \emph{contacted} creature\\
    \textbf{Saving Throw:}
    Will partial\\
    \textbf{Cost:}
    \psionic{contact}\\
    \textbf{Maintenance Cost:}
    4 pp/round\\
    \textbf{Prerequisites:}
    \psionic{mindlink}\\
}
{
    As the \spell{fear} spell, except as noted here.

    If the Will save succeeds, the creature is shaken for 1 round and the power does not need to be maintained.
}
