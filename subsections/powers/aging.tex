\Psionic{Aging}{aging}
{Psychometabolism [Evil]}
{
	\textbf{Level:}
	Egoist 3\\
	\textbf{Manifesting Time:}
	1 mental action\\
	\textbf{Range:}
	Touch\\
	\textbf{Target:}
	Living creature touched\\
	\textbf{Saving Throw:}
	Fortitude half\\
	\textbf{Cost:}
	15 power points\\
	\textbf{Maintenance Cost:}
	None (Instantaneous)\\
	% \textbf{Critical Success:}
	% The subject ages 2d10 years.\\
	% \textbf{Critical Failure:}
	% You age 1d10 years.\\
}
{
	You cause unnatural aging by touch. The subject ages 2d4 years. If the creature successfully saves, they age half this amount.
}
