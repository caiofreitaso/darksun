\Spell{Blindscorch}{blindscorch}
{Evocation [Fire]}
{
	\textbf{Level:}
	Drought 5\\
	\textbf{Components:}
	V, S, DF\\
	\textbf{Casting Time:}
	1 standard action\\
	\textbf{Range:}
	Medium (30 m + 3 m/level)\\
	\textbf{Target:}
	One creature\\
	\textbf{Duration:}
	Instantaneous\\
	\textbf{Saving Throw:}
	Fortitude partial\\
	\textbf{Spell Resistance:}
	Yes\\
}
{
	You burn the face of one opponent. The subject takes 1d4 points of fire damage per level (maximum 10d4) and is permanently blinded. A successful Fortitude save halves the damage and negates the blindness.
}
