\Spell{Echo of the Lirr}{echo of the lirr}
{Evocation [Sonic]}
{
	\textbf{Level:}
	Drd 2, Rgr 2\\
	\textbf{Components:}
	V, S, F\\
	\textbf{Casting Time:}
	1 standard action\\
	\textbf{Range:}
	9 meters\\
	\textbf{Area:}
	Cone-shaped burst\\
	\textbf{Duration:}
	Instantaneous (see text)\\
	\textbf{Saving Throw:}
	Fortitude negates\\
	\textbf{Spell Resistance:}
	Yes\\
}
{
	Creatures caught in the cone of sound you emit must make a successful Fortitude save or be stunned for 1d3 rounds. Creatures that cannot hear are not stunned.

	Focus: A lirr scale.
}
